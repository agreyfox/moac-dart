String getSelection ()

Get the selected parameter as a string

Implementation 

String getSelection() {
  if (_latest) {
    return ethLatest;
  }
  if (_earliest) {
    return ethEarliest;
  }
  if (_pending) {
    return ethPending;
  }
  if (_number != null) {
    return MoacUtilities.intToHex(_number);
  }
  return null;
}
